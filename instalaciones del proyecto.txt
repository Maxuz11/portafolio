import Swal from 'sweetalert2';
@Component({
  selector: 'app-my-component',
  template: `
    <button (click)="showAlert()">Show Alert</button>
  `,
})
export class MyComponent {
  showAlert() {
    Swal.fire('Hello!', 'This is a SweetAlert2 alert.');
  }
}
